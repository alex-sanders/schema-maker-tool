JSON-LD

Note: If you are closed on Sunday or weekends, you will have to remove a comma after the last OpeningHoursSpecification's object.


  <script type="application/ld+json">
  {
    "@context": "http://schema.org/",
    "@type": "Event",
    "name": "{{name}}",
    "description": "{{description}}",
    "image": "{{image}}",
    "url": "{{url}}",
    "startDate": "{{yearstart}}-{{monthstart}}-{{daystart}}T{{hourstart}}:{{minutestart}}",
    "endDate": "{{yearend}}-{{monthend}}-{{dayend}}",
    "location": {
      "@type": "Place",
      "name": "{{namelocation}}",
      "address": {
        "@type": "PostalAddress",
        "streetAddress": "{{streetaddress}}",
        "addressLocality": "{{addresslocality}}",
        "addressRegion": "{{addressregion}}",
         "postalCode": "{{postalcode}}",
        "addressCountry": "{{addresscountry}}"
      }
    },
    "performer": {
      "@type": "PersonPerformingGroup",
      "name": "{{nameperformer}}"
    },
    "offers": {
      "@type": "Offer",
      "priceCurrency": "USD",
      "url": "{{urloffer}}",
      "availability": "http://schema.org/Discontinued",
      "availability": "http://schema.org/InStock",
      "availability": "http://schema.org/InStoreOnly",
      "availability": "http://schema.org/LimitedAvailability",
      "availability": "http://schema.org/OnlineOnly",
      "availability": "http://schema.org/OutOfStock",
      "availability": "http://schema.org/PreOrder",
      "availability": "http://schema.org/PreSale",
      "availability": "http://schema.org/SoldOut",
      "validFrom": "{{yearvalid}}-{{monthvalid}}-{{datevalid}}",
      "price": "{{price}}"
    }
 }
  </script>
