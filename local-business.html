LocalBusiness Schema

Fill in the textbox below. The JSON-LD and Microdata below will update. Then copy the text and paste the following code into Google's Stuctured Data Testing Tool!

Replace "LocalBusiness" in the markup with any more specific item types (ex. Restaurant, HealthClub, Dentist, FinancialService, etc.). You can find more specific item types nested under LocalBusiness in Schema.org's Full Schema Heirarchy



JSON-LD

Note: If you are closed on Sunday or weekends, you will have to remove a comma after the last OpeningHoursSpecification's object.


  <script type="application/ld+json">
  {
    "@context": "http://schema.org/",
    "@type": "LocalBusiness",
    "name": "{{name}}",
    "description": "{{description}}",
    "telephone": "{{telephone}}",
    "image": "{{image}}",
    "priceRange": "{{pricerange}}",
    "aggregateRating": {
      "@type": "AggregateRating",
      "ratingValue": "{{ratingvalue}}",
      "reviewCount": "{{ratingcount}}",
      "bestRating": "{{bestrating}}"
    },
    "hasMap": "{{hasmap}}",
    "geo": {
      "@type": "GeoCoordinates",
      "latitude": "{{latitude}}",
      "longitude": "{{longitude}}"
    },
    "openingHoursSpecification": [{
      "@type": "OpeningHoursSpecification",
      "dayOfWeek": ["Monday","Tuesday","Wednesday","Thursday","Friday"],
      "opens": "{{opensweekday}}",
      "closes": "{{closesweekday}}"
    },{
      "@type": "OpeningHoursSpecification",
      "dayOfWeek": "Monday",
      "opens": "{{opensmonday}}",
      "closes": "{{closesmonday}}"
    }, {
      "@type": "OpeningHoursSpecification",
      "dayOfWeek": "Tuesday",
      "opens": "{{openstuesday}}",
      "closes": "{{closestuesday}}"
    }, {
      "@type": "OpeningHoursSpecification",
      "dayOfWeek": "Wednesday",
      "opens": "{{openswednesday}}",
      "closes": "{{closeswednesday}}"
    }, {
      "@type": "OpeningHoursSpecification",
      "dayOfWeek": "Thursday",
      "opens": "{{opensthursday}}",
      "closes": "{{closesthursday}}"
    },{
      "@type": "OpeningHoursSpecification",
      "dayOfWeek": "Friday",
      "opens": "{{opensfriday}}",
      "closes": "{{closesfriday}}"
    },{
      "@type": "OpeningHoursSpecification",
      "dayOfWeek": "Saturday",
      "opens": "{{openssaturday}}",
      "closes": "{{closessaturday}}"
    },{
      "@type": "OpeningHoursSpecification",
      "dayOfWeek": "Sunday",
      "opens": "{{openssunday}}",
      "closes": "{{closessunday}}"
    }{
      "@type": "OpeningHoursSpecification",
      "dayOfWeek": ["Saturday","Sunday"],
      "opens": "{{opensweekend}}",
      "closes": "{{closesweekend}}"
    }],
    "address": {
      "@type": "PostalAddress",
      "streetAddress": "{{streetaddress}}",
      "addressLocality": "{{addresslocality}}",
      "addressRegion": "{{addressregion}}",
      "postalCode": "{{postalcode}}",
      "addressCountry": "{{addresscountry}}"
    }
 }
  </script>
