JSON-LD

Note: If you are closed on Sunday or weekends, you will have to remove a comma after the last OpeningHoursSpecification's object.


  <script type="application/ld+json">
  {
    "@context": "http://schema.org/",
    "@type": "Article",
    "headline": "{{headline}}",
    "description": "{{description}}",
    "mainEntityOfPage": {
      "@type": "WebPage",
      "@id": "{{url}}"
    },
    "image": {
      "@type": "ImageObject",
      "url": "{{image}}",
      "height": "{{heightimage}}",
      "width": "{{widthimage}}"
    },
    "datePublished": "{{yearpublished}}-{{monthpublished}}-{{daypublished}}",
    "dateModified": "{{yearmodified}}-{{monthmodified}}-{{daymodified}}",
    "author": {
      "@type": "PersonOrganization",
      "name": "{{nameauthor}}"
    },
    "publisher": {
      "@type": "Organization",
      "name": "{{namepublisher}}",
      "logo": {
        "@type": "ImageObject",
        "url": "{{imagelogo}}",
        "height": "{{heightlogo}}",
        "width": "{{widthlogo}}"
      }
    }
 }
  </script>
